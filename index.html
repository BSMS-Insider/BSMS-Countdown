<script>

// =======================

// CONFIGURATION

// =======================

const MANUAL_NO_SCHOOL = false; // set to true to manually show "No School"

const events = [

  "07:50", // Auditorium → Dismissal

  "07:55", // Dismissal → 1st

  "08:00", // 1st Period start

  "08:50", // 1st Period end → Class change

  "08:55", // 2nd Period start

  "09:45", // 2nd end → Class change

  "09:50", // 3rd start

  "10:40", // 3rd end → Class change

  "10:45", // 4th Pt.1 start

  "11:25", // 4th Pt.1 end → Lunch

  "11:50", // 4th Pt.2 start

  "12:25", // 4th Pt.2 end → Class change

  "12:30", // 5th start

  "13:20", // 5th end → Class change

  "13:25", // 6th start

  "14:15", // 6th end → Class change

  "14:20", // 7th start

  "15:10"  // 7th end → School ends

];


// =======================

// FUNCTIONS

// =======================

function updateCountdown() {

  const now = new Date();


  // MANUAL OVERRIDE

  if (MANUAL_NO_SCHOOL) {

    document.getElementById('label').textContent = "No School";

    document.getElementById('timer').textContent = "";

    return;

  }


  const day = now.getDay(); // 0=Sun, 6=Sat

  if (day === 0 || day === 6) {

    document.getElementById('label').textContent = "Weekend";

    document.getElementById('timer').textContent = "";

    return;

  }


  // Find next event

  let nextEvent = null;

  for (let time of events) {

    const [h, m] = time.split(":").map(Number);

    const eventTime = new Date(now);

    eventTime.setHours(h, m, 0, 0);

    if (eventTime > now) {

      nextEvent = eventTime;

      break;

    }

  }


  if (!nextEvent) {

    document.getElementById('label').textContent = "School has ended.";

    document.getElementById('timer').textContent = "";

    return;

  }


  // Countdown

  const diff = nextEvent - now;

  const mins = Math.floor(diff / 60000);

  const secs = Math.floor((diff % 60000) / 1000);


  document.getElementById('label').textContent = "Next class change / period start in";

  document.getElementById('timer').textContent = `${mins}:${secs.toString().padStart(2,"0")}`;

}


// =======================

// INIT

// =======================

updateCountdown();

setInterval(updateCountdown, 1000);


</script>
